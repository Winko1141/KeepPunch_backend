#### 关键代码

```kotlin
class HomeFragment : Fragment() {
    private var _binding: FragmentHomeBinding? = null
    private val binding get() = _binding!!
    private var selectedDate = LocalDate.now()
    private val today = LocalDate.now()
    private val apiService = RetrofitClient.apiService
    private lateinit var loadingStateHelper: LoadingStateHelper
    
    private var categories: List<CategoryItem> = emptyList()
    private var currentCategoryFilter: String? = null
    private var currentCategoryName = "全部"
    
    override fun onCreateView(...): View {
        _binding = FragmentHomeBinding.inflate(inflater, container, false)
        
        // 初始化状态管理
        loadingStateHelper = LoadingStateHelper(
            contentView = binding.scrollContent,
            loadingView = binding.layoutLoading.root,
            emptyView = binding.layoutEmpty.root,
            errorView = binding.layoutError.root
        )
        
        setupCalendar()
        setupTaskList()
        
        // 添加任务按钮
        binding.fabAdd.setOnClickListener {
            findNavController().navigate(R.id.action_navigation_home_to_addTaskFragment)
        }
        
        setupCategoryFilter()
        loadCategories()
        loadTasks(selectedDate)
        
        return binding.root
    }
    
    private fun setupCalendar() {
        val calendarView = binding.calendarView
        val today = LocalDate.now()
        val startDate = today.minusMonths(12)
        val endDate = today.plusMonths(12)
        
        calendarView.setup(startDate, endDate, DayOfWeek.MONDAY)
        calendarView.scrollToDate(today)
        
        calendarView.dayBinder = object : WeekDayBinder<DayViewContainer> {
            override fun create(view: View) = DayViewContainer(view)
            
            override fun bind(container: DayViewContainer, data: WeekDay) {
                val date = data.date
                
                // 设置周几文字
                val weekText = if (date == today) "今天" else {
                    when (date.dayOfWeek) {
                        DayOfWeek.MONDAY -> "周一"
                        // ...
                        else -> "周日"
                    }
                }
                container.tvWeek.text = weekText
                container.tvDate.text = "${month}.${day}"
                
                // 根据选中状态设置样式
                if (date == selectedDate) {
                    container.root.setBackgroundResource(R.drawable.bg_calendar_selected)
                    container.tvWeek.setTextColor(Color.WHITE)
                    container.tvDate.setTextColor(Color.WHITE)
                } else if (date == today) {
                    container.root.setBackgroundResource(R.drawable.bg_calendar_today)
                    container.tvWeek.setTextColor(Color.ORANGE)
                    container.tvDate.setTextColor(Color.ORANGE)
                }
                
                // 点击切换日期
                container.root.setOnClickListener {
                    val oldDate = selectedDate
                    selectedDate = date
                    calendarView.notifyDateChanged(oldDate)
                    calendarView.notifyDateChanged(date)
                    loadTasks(date)
                }
            }
        }
    }
    
    private fun loadTasks(date: LocalDate) {
        val dateStr = date.format(DateTimeFormatter.ISO_LOCAL_DATE)
        lifecycleScope.launch {
            loadingStateHelper.showLoading()
            try {
                val response = apiService.getTaskList(dateStr, currentCategoryFilter)
                if (response.isSuccessful) {
                    val body = response.body()
                    if (body?.code == 200) {
                        val tasks = body.data ?: emptyList()
                        
                        // 调度提醒
                        tasks.forEach { task ->
                            if (task.reminder_enabled && !task.reminder_time.isNullOrEmpty()) {
                                NotificationHelper.cancelTaskReminder(requireContext(), task.id)
                                NotificationHelper.scheduleTaskReminder(requireContext(), task)
                            }
                        }
                        
                        if (tasks.isEmpty()) {
                            loadingStateHelper.showEmpty(
                                message = "今天还没有任务",
                                hint = "点击右下角按钮添加你的第一个目标"
                            )
                        } else {
                            updateTaskList(tasks)
                            loadingStateHelper.showContent()
                        }
                    } else {
                        loadingStateHelper.showError(message = "加载失败", 
                            onRetry = { loadTasks(date) })
                    }
                }
            } catch (e: Exception) {
                val errorMsg = ErrorHandler.handleException(e, requireContext())
                loadingStateHelper.showError(message = errorMsg, 
                    onRetry = { loadTasks(date) })
            }
        }
    }
    
    private fun updateTaskList(tasks: List<TaskItem>) {
        val incomplete = tasks.filter { !it.is_finished }
        val complete = tasks.filter { it.is_finished }
        
        binding.rvIncomplete.adapter = TaskAdapter(
            incomplete, lifecycleScope,
            onTaskClick = { task -> navigateToDetail(task) },
            onCheckinClick = { task -> handleCheckin(task, false) }
        )
        
        binding.rvCompleted.adapter = TaskAdapter(
            complete, lifecycleScope,
            onTaskClick = { task -> navigateToDetail(task) },
            onCheckinClick = { task -> handleCheckin(task, true) }
        )
    }
    
    private fun handleCheckin(task: TaskItem, isAlreadyChecked: Boolean) {
        if (!isAlreadyChecked && selectedDate.isAfter(today)) {
            Toast.makeText(requireContext(), "不能打卡未来日期", Toast.LENGTH_SHORT).show()
            return
        }
        
        val dateStr = selectedDate.format(DateTimeFormatter.ISO_LOCAL_DATE)
        val request = CheckinRequest(task.id, dateStr)
        
        lifecycleScope.launch {
            try {
                val response = if (isAlreadyChecked) {
                    apiService.cancelCheckin(request)
                } else {
                    apiService.checkin(request)
                }
                
                if (response.isSuccessful && response.body()?.code == 200) {
                    loadTasks(selectedDate)  // 刷新列表
                } else {
                    Toast.makeText(requireContext(), 
                        response.body()?.msg ?: "操作失败", Toast.LENGTH_SHORT).show()
                }
            } catch (e: Exception) {
                Toast.makeText(requireContext(), "网络错误", Toast.LENGTH_SHORT).show()
            }
        }
    }
}
```

```kotlin
class WeekStatsFragment : Fragment(R.layout.fragment_week_stats) {
    private var currentDate: LocalDate = LocalDate.now()
    private val today: LocalDate = LocalDate.now()
    private val apiService = RetrofitClient.apiService
    
    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        val recyclerView = view.findViewById<RecyclerView>(R.id.rvWeekStat)
        val tvWeekRange = view.findViewById<TextView>(R.id.tv_week_range)
        val ivPrev = view.findViewById<ImageView>(R.id.iv_prev_week)
        val ivNext = view.findViewById<ImageView>(R.id.iv_next_week)
        
        recyclerView.layoutManager = LinearLayoutManager(requireContext())
        
        // 上一周
        ivPrev.setOnClickListener {
            currentDate = currentDate.minusWeeks(1)
            updateWeekRange(tvWeekRange)
            loadWeekData(currentDate, recyclerView)
            updateArrowState(ivNext)
        }
        
        // 下一周（不能超过当前周）
        ivNext.setOnClickListener {
            if (!isCurrentWeek()) {
                currentDate = currentDate.plusWeeks(1)
                updateWeekRange(tvWeekRange)
                loadWeekData(currentDate, recyclerView)
                updateArrowState(ivNext)
            }
        }
    }
    
    private fun updateWeekRange(tv: TextView) {
        val startOfWeek = currentDate.with(DayOfWeek.MONDAY)
        val endOfWeek = currentDate.with(DayOfWeek.SUNDAY)
        val formatter = DateTimeFormatter.ofPattern("MM/dd")
        tv.text = "${startOfWeek.format(formatter)} ~ ${endOfWeek.format(formatter)}"
    }
    
    private fun calculateWeeklyTarget(repeatType: Int, weekRule: String, targetCount: Int): Int {
        return when (repeatType) {
            1 -> if (weekRule.isEmpty()) 7 else weekRule.split(",").size
            2 -> if (targetCount > 0) targetCount else 7
            else -> 7
        }
    }
}
```

```kotlin
private fun updateAdapter() {
    val items = mutableListOf<TaskManagerItem>()
    
    // 分类部分
    if (categories.isNotEmpty()) {
        items.add(TaskManagerItem.SectionHeader("分类管理", categories.size, showAddButton = true))
        categories.forEach { category ->
            items.add(TaskManagerItem.CategoryItemWrapper(category))
        }
    }
    
    // 任务部分（按分类分组）
    items.add(TaskManagerItem.SectionHeader("全部任务", allTasks.size, showAddButton = false))
    
    val tasksByCategory = allTasks.groupBy { it.category_id }
    
    categories.forEach { category ->
        val categoryTasks = tasksByCategory[category.id] ?: emptyList()
        categoryTasks.forEach { task ->
            items.add(TaskManagerItem.TaskItemWrapper(task))
        }
    }
    
    // 未分类任务
    val uncategorizedTasks = tasksByCategory.filterKeys { it == null }.values.flatten()
    uncategorizedTasks.forEach { task ->
        items.add(TaskManagerItem.TaskItemWrapper(task))
    }
    
    binding.rvCategories.adapter = CategoryAdapter(items, lifecycleScope, ...)
}
```

### 10.1 AlarmReceiver.kt

文件路径：[AlarmReceiver.kt](file:///d:/xm/KeepPunch_frontend/app/src/main/java/com/example/keeppunch/receiver/AlarmReceiver.kt)

```kotlin
class AlarmReceiver : BroadcastReceiver() {
    override fun onReceive(context: Context, intent: Intent) {
        val taskId = intent.getIntExtra(NotificationHelper.EXTRA_TASK_ID, 0)
        val taskName = intent.getStringExtra(NotificationHelper.EXTRA_TASK_NAME) ?: "任务提醒"
        val taskIcon = intent.getStringExtra(NotificationHelper.EXTRA_TASK_ICON) ?: "book"
        
        // 创建通知渠道
        NotificationHelper.createNotificationChannel(context)
        
        // 构建通知
        val notificationBuilder = NotificationCompat.Builder(
            context, 
            NotificationHelper.CHANNEL_ID
        )
            .setSmallIcon(IconMapper.getResIdFromName(taskIcon))
            .setContentTitle(taskName)
            .setContentText("该完成你的任务了！")
            .setPriority(NotificationCompat.PRIORITY_HIGH)
            .setAutoCancel(true)
        
        // 点击通知打开应用
        val launchIntent = context.packageManager
            .getLaunchIntentForPackage(context.packageName)
        val pendingIntent = PendingIntent.getActivity(
            context, taskId, launchIntent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )
        notificationBuilder.setContentIntent(pendingIntent)
        
        // 显示通知
        val notificationManager = context.getSystemService(
            Context.NOTIFICATION_SERVICE
        ) as NotificationManager
        notificationManager.notify(taskId, notificationBuilder.build())
    }
}
```
